class User {
  #password;

  constructor(name, password) {
    this.name = name;
    this.#password = password;
  }

  checkPassword(input) {
    return this.#password === input;
  }
}

const u = new User("Alex", "1234");
console.log(u.checkPassword("1234")); // true
