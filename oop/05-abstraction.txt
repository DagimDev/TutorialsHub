class Shape {
  constructor() {
    if (new.target === Shape) {
      throw new Error("Cannot instantiate abstract class Shape");
    }
  }

  area() {
    throw new Error("Must implement area method");
  }
}

class Square extends Shape {
  constructor(side) {
    super();
    this.side = side;
  }

  area() {
    return this.side * this.side;
  }
}

const s = new Square(4);
console.log(s.area()); // 16
