2. âœ… Sum of Digits
sumDigits(1234) => 1 + 2 + 3 + 4 = 10

function sumDigits(n) {
  if (n === 0) return 0;
  return (n % 10) + sumDigits(Math.floor(n / 10));
}

3. âœ… Reverse String

function reverse(str) {
  if (str === "") return "";
  return reverse(str.slice(1)) + str[0];
}
